library(animation)
library(fields)
library(spam)
library(MBA)
library(sp)

library(RPostgreSQL)
library(DBI)
library(cclust)
library(plyr)
library(RJSONIO)
library(RCurl)
library(bitops)
library(spTimer)
library(coda)
library(lattice)
par(mar = c(4, 4, 0.5, 0.5))
ani.options(interval = 0.5)
for (hour in 1:100) {
    plot(union.coords, xlab = "Longitude", ylab = "Latitude")
    surf <- rbind(cbind(site.coords, true.val[hour, ]), cbind(grid.coords, 
        grid.val[hour, ]))
    surf <- mba.surf(surf, 200, 200)$xyz
    brk <- seq(0, 0.02, by = (0.02 + 0)/(100 - 1))
    image.plot(surf, breaks = brk, col = tim.colors(99), 
        xlab = "Longitude", ylab = "Latitude", axes = F, 
        add = TRUE)
    contour(surf, nlevels = 15, lty = 3, add = T)
    ani.pause()
}
